def extract_begin

	if (@head == nil)
		"No se puede extraer un nodo, la lista se encuentra vacía"
	else
		valor = @head.value

		if @head == @tail
			@head,@tail = nil

		else
			@head = @head[:next]
			@head[:prev] = nil

		end
		valor
	end

end

def insert_begin(value)

	if (@head == nil)

		@head = @Node.new(value,nil,nil)
		@tail = @head

	else

		@head[:prev] = @Node.new(value,@head,nil)
		@head = @head[:prev] 

	end

end
